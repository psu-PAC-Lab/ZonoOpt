339 // ---- Program-wide default solver settings -------------------------------
340 //
341 // ZonoOpt's optimization methods take a SolverSettings argument whose default
342 // is whatever set_default_solver_settings() last installed. This lets a user
343 // switch the entire library to (e.g.) Gurobi by calling
344 // set_default_solver_settings(GurobiSettings{});
345 // once at program startup, instead of passing GurobiSettings to every call.
346 //
347 // Initial default is OptSettings() (internal ADMM/branch-and-bound).
348 //
349 // Not thread-safe: set_default_solver_settings should be called from a single
350 // thread at startup before parallel optimization begins. Existing const
351 // SolverSettings& references obtained from get_default_solver_settings() are
352 // invalidated by a subsequent set_default_solver_settings() call.
353
354 namespace detail
355 {
356 inline std::unique_ptr<SolverSettings>& default_settings_storage()
357 {
358 static std::unique_ptr<SolverSettings> p = std::make_unique<OptSettings>();
359 return p;
360 }
361 } // namespace detail
362
368 inline const SolverSettings& get_default_solver_settings()
369 {
370 return *detail::default_settings_storage();
371 }
372
383 inline void set_default_solver_settings(const SolverSettings& settings)
384 {
385 settings.verify_available();
386 detail::default_settings_storage() = settings.clone();
387 }
