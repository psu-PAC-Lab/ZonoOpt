171 void inverse()
172 {
173 const zono_float min = y_min();
174 const zono_float max = y_max();
175
176 if (std::abs(min) < zono_eps && std::abs(max) < zono_eps) // empty set
177 {
178 y_min() = std::numeric_limits<zono_float>::quiet_NaN();
179 y_max() = std::numeric_limits<zono_float>::quiet_NaN();
180 }
181 else if (min > zono_eps || max < -zono_eps)
182 {
183 y_min() = one / max;
184 y_max() = one / min;
185 }
186 else if (std::abs(min) < zono_eps && max > zono_eps)
187 {
188 y_min() = one / max;
189 y_max() = std::numeric_limits<zono_float>::infinity();
190 }
191 else if (min < -zono_eps && std::abs(max) < zono_eps)
192 {
193 y_min() = -std::numeric_limits<zono_float>::infinity();
194 y_max() = one / min;
195 }
196 else // y_min < 0, y_max > 0
197 {
198 y_min() = -std::numeric_limits<zono_float>::infinity();
199 y_max() = std::numeric_limits<zono_float>::infinity();
200 }

277 void sin_assign(const Derived& x)
278 {
279 if (x.y_max() - x.y_min() >= two * pi)
280 {
281 y_max() = one;
282 y_min() = -one;
283 }
284 else
285 {
286 // shift domain to [-pi, pi]
287 zono_float u = x.y_max(); // init
288 zono_float l = x.y_min(); // init
289 while (u > pi)
290 {
291 u -= two * pi;
292 l -= two * pi;
293 }
294 while (l < -pi)
295 {
296 u += two * pi;
297 l += two * pi;
298 }
299
300 // get bounds
301 y_max() = ((l < pi / two && pi / two < u) || (l < -3 * pi / two && -3 * pi / two < u))
302 ? one
303 : std::max(std::sin(u), std::sin(l));
304 y_min() = ((l < -pi / two && -pi / two < u) || (l < 3 * pi / two && 3 * pi / two < u))
305 ? -one
306 : std::min(std::sin(u), std::sin(l));
307 }
308 }
