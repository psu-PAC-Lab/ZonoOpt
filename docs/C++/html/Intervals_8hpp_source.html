249 void sin_assign(const Derived& x)
250 {
251 if (x.y_max() - x.y_min() >= two * pi)
252 {
253 y_max() = one;
254 y_min() = -one;
255 }
256 else
257 {
258 // shift domain to [-pi, pi]
259 zono_float u = x.y_max(); // init
260 zono_float l = x.y_min(); // init
261 while (u > pi)
262 {
263 u -= two * pi;
264 l -= two * pi;
265 }
266 while (l < -pi)
267 {
268 u += two * pi;
269 l += two * pi;
270 }
271
272 // get bounds
273 y_max() = ((l < pi / two && pi / two < u) || (l < -3 * pi / two && -3 * pi / two < u))
274 ? one
275 : std::max(std::sin(u), std::sin(l));
276 y_min() = ((l < -pi / two && -pi / two < u) || (l < 3 * pi / two && 3 * pi / two < u))
277 ? -one
278 : std::min(std::sin(u), std::sin(l));
279 }
